    'Scan all files in folder 'Media'
    ChDir "Media"

    StartSearch

        Repeat
            file$=FindFile()
            
            If file$="" Then Exit 'search has ended
            
            fsize=FileSize(file$) 'figure out the file size
            
            Print file$+", "+fsize+" bytes" 'print info
        Forever

    EndSearch

    WaitKey
